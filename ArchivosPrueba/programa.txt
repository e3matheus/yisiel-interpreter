var x, r : value

proc poten(in base, in exp, out r) as
var a : value
begin
a <- 1;
if exp=0 -> skip
   exp % 2 = 0 -> poten(base*base, exp/2, a)
   exp % 2 != 0 -> base*poten(base, exp-1, a)
fi;
r <- a;
return;
end

main
x <- 4;
poten(x, 2, r);
do r>0 -> 
	begin
	x <- x-1;
	r <- r-1;
	end
od;
show r;
show x;
end
